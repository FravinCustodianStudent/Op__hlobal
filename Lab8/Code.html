class Range {
public:
    Range() {
        this->startOfRange = NULL;
        this->EndOfRange = NULL;
        this->accuracy = NULL;
        this->result = NULL;
    }
    Range(float startOfRange) {
        this->startOfRange = startOfRange;
        this->EndOfRange = NULL;
        this->accuracy = NULL;

    }
    Range(float startOfRange, float EndOfRange) :Range(startOfRange) {
        this->EndOfRange = EndOfRange;
    }
    Range(float startOfRange, float EndOfRange, float accuracy) :Range(startOfRange, EndOfRange) {
        this->accuracy = accuracy;
    }
    float Get_start() {
        return startOfRange;
    }
    float Get_end() {
        return EndOfRange;
    }
    float Get_acc() {
        return accuracy;
    };
    void Set_start(float startOfRange) {
        this->startOfRange = startOfRange;
    }
    void Set_end(float EndOfRange) {
        this->EndOfRange = EndOfRange;
    }
    void Set_acc(float accuracy) {
        this->accuracy = accuracy;
    }
    float Get_result() {
        return result;
    }
    void Set_result(float result) {
        this->result = result;
    }
private:
    float result;
    float startOfRange;
    float EndOfRange;
    float accuracy;
};

void find_root(Range *range) {
    float h,f_x,f_1x,x;
    cout << "Original function: x^2+4sin(x)\nDerivative of a function:2x+4cos(x)" << endl;
    x = range->Get_start();
    do
    {

        f_x = pow(x, 2) + 4 * sin(x);
        f_1x = 2 * x + 4 * cos(x);
        h = f_x / f_1x;
        x -= h;

    } while (fabs(h)>range->Get_acc());
    cout << "Result: " << x << endl;
    range->Set_result(x);


}
