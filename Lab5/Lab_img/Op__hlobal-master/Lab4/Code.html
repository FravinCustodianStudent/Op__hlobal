int factorial(int N) {
    if (N <= 0)
    {
        return 1;
    }
    else if (N == 1) {
        return 1;
    }
    return N * factorial(N - 1);
}
double ln(float x) {
    float sum = x - 1;
    for (int i = 2; i < 1000; i++)
    {
        sum+=(pow((x - 1), i) / i)* pow(-1, i + 1);
    }
    return sum;
}
