void task1() {
    const int amount = 3;
    int set, result[amount], x[amount], y[amount], z[amount];
    cout << "\n----TASK1----" << endl;
    for (int i = 0; i < amount; i++)
    {

        cout << "set x[ " << i << " ] : ";
        while (true) {
            cin >> set;
            if (set==1|| set==0)
            {
                x[i] = set;
                break;
            }
            else {
                cout << "x can be only 1 or 0 : ";
            }
        }
        cout << "set y[ " << i << " ] : ";
        while (true) {
            cin >> set;
            if (set == 1 || set == 0)
            {
                y[i] = set;
                break;
            }
            else {
                cout << "x can be only 1 or 0 : ";
            }
        }
        cout << "set z[ " << i << " ] : ";
        while (true) {
            cin >> set;
            if (set == 1 || set == 0)
            {
                z[i] = set;
                break;
            }
            else {
                cout << "x can be only 1 or 0 : ";
            }
        }
        if (x[i] == y[i] && y[i] == z[i] && x[i]==1) {
            result[i] = 1;
        }
        else if(x[i]== y[i] && z[i]==0 && x[i]==1){
            result[i] = 1;
        }
        else {
            result[i] = 0;
        }
        cout << "\n--------\n";


    }
    cout << "--------\n | x | y | z | result |" << endl;

    for (int i = 0; i < amount; i++)
    {
        cout <<" | "<< x[i] << " | " << y[i] << " | " << z[i] << " |    " << result[i] << "   |"<< endl;
    }


}
