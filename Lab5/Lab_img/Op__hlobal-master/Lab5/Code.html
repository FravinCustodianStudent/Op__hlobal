int factorial(long long N) {
	if (N <= 0)
	{
		return 1;
	}
	else if (N == 1) {
		return 1;
	}
	return N * factorial(N - 1);
}
int akkerman(int m,int n,int &deep) {
    if (m==0)
    {
        return n+1;
    }
    else if(m>0 && n==0){
        akkerman(m - 1, 1,++deep);
    }
    else if (m > 0 && n > 0) {
        akkerman(m - 1, akkerman(m, n - 1, ++deep),++deep);

    }
}
